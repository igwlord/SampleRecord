<script>
        // Generate decorative waveform bars
        const container = document.getElementById('waveform-container');
        for (let i = 0; i < 80; i++) {
            const bar = document.createElement('div');
            bar.className = 'waveform-bar';
            bar.style.height = (Math.random() * 80 + 10) + '%';
            container.appendChild(bar);
        }

        // Animate bars during recording
        function animateWaveform() {
            const bars = document.querySelectorAll('.waveform-bar');
            bars.forEach(bar => {
                const newHeight = Math.random() * 80 + 10;
                bar.style.height = newHeight + '%';
                bar.style.opacity = (newHeight / 100) + 0.2;
            });
        }

        let recordingInterval;
        let isRecording = false;

        function toggleRecording() {
            isRecording = !isRecording;
            const idleView = document.getElementById('idle-view');
            const recordingView = document.getElementById('recording-view');
            const contextGrid = document.getElementById('context-grid');

            if (isRecording) {
                idleView.classList.add('opacity-0', 'scale-90', 'pointer-events-none');
                setTimeout(() => {
                    idleView.classList.add('hidden');
                    recordingView.classList.remove('hidden');
                    setTimeout(() => recordingView.classList.remove('opacity-0'), 50);
                }, 500);
                
                contextGrid.classList.add('opacity-20');
                recordingInterval = setInterval(animateWaveform, 100);
            } else {
                recordingView.classList.add('opacity-0');
                setTimeout(() => {
                    recordingView.classList.add('hidden');
                    idleView.classList.remove('hidden');
                    setTimeout(() => idleView.classList.remove('opacity-0', 'scale-90', 'pointer-events-none'), 50);
                }, 500);
                
                contextGrid.classList.remove('opacity-20');
                clearInterval(recordingInterval);
            }
        }
    </script>
